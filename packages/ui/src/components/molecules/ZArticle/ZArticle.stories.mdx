<Preview>
  <Story name="Grid">
    {{
        components: { ZArticle },
        data() {
          return {
            posts: [],
          }
        },
        template: `<div style="display: grid; grid-template-columns: repeat(4, 1fr); column-gap: 20px; row-gap: 20px; max-width: 1071px; margin: 0 auto">
          <ZArticle
            v-for="(post, index) in posts"
            :key="post.id"
            :thumbnail="'https://demo.przemyslawspaczek.pl/wp-content/uploads/'+post.media.file"
            :title="post.title.rendered"
            :to="post.link"
            author="Przemysław Spaczek"
            :sticky="post.sticky"
            :date="post.date"
            :class="{'z-article--highlighted': post.sticky, 'z-article--primary': index === 0}"
          />
        </div>`,
      async created(){
        const { API_URL } = process.env;
        const params = {
          per_page: 16,
        };
        const { data } = await axios.get(`${API_URL}/posts`, { params });
        this.posts = data;
      }
    }}
  </Story>
